CREATE DEFINER=`sshilpika`@`localhost` PROCEDURE `findPatientsByPharmacistId`(IN `pharmacistId` VARCHAR(9)) NOT DETERMINISTIC NO SQL SQL SECURITY DEFINER BEGIN SELECT person.SSN, person.First_Name, person.Last_Name, person.Birth_Date, person.Gender, patient.Current_Status FROM Person person, Patient patient, Has_Patients hp WHERE hp.PH_SSN = pharmacistId AND hp.P_SSN = person.SSN AND patient.SSN = person.SSN; END
